BoxJS Game Framework

Simple but powerful JavaScript Videogame Framework

import { * } from 'box-js.js';

const game = new BOXJS.box(FLAGS.FILL_SCREEN, [1280,720], [0,0]);

game.anchor = ANCHORS.CENTER; //TOP_RIGHT, TOP_LEFT, BOTTOM_RIGHT, BOTTOM_LEFT, CENTER_TOP, CENTER_RIGHT, CENTER_BOTTOM, CENTER_LEFT, CENTER

game.create_scene(0, "main_menu", new BOXJS.scene([
  new BOXJS.Sprite(SPRITES.SQUARE_GREEN, BOXJS.Vec2(0,0), 0, spinnySquare)
], COLORS.DARK_GRAY));


function spinnySquare(sprite){
  sprite.rotation += 2
}
